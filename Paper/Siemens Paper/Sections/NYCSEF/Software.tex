\begin{algorithm}[!ht]
	\caption{FizzBuzz Algorithm}
	\begin{algorithmic}[1]
		\For{each integer $i$ 1 to 100}
			\If{$15 \mid i$}
				\State print ``FizzBuzz''
			\ElsIf{$3 \mid i$}
				\State print ``Fizz''
			\ElsIf{$5 \mid i$}
				\State print ``Buzz''
			\Else
				\State print $i$
			\EndIf
		\EndFor
	\end{algorithmic}
\end{algorithm}
